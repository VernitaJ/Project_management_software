019package org.apache.commons.compress.compressors.snappy;
020
021/**
022 * Dialects of the framing format that {@link FramedSnappyCompressorInputStream} can deal with.
023 * @since 1.12
024 */
025public enum FramedSnappyDialect {
026    /**
027     * The standard as defined by the <a
028     * href="https://github.com/google/snappy/blob/master/framing_format.txt">Snappy
029     * framing format description</a>
030     */
031    STANDARD(true, true),
032    /**
033     * The format used by Apple's iWork Archives (.iwa files).
034     */
035    IWORK_ARCHIVE(false, false);
036
037    private final boolean streamIdentifier, checksumWithCompressedChunks;
038
039    FramedSnappyDialect(final boolean hasStreamIdentifier,
040                        final boolean usesChecksumWithCompressedChunks) {
041        this.streamIdentifier = hasStreamIdentifier;
042        this.checksumWithCompressedChunks = usesChecksumWithCompressedChunks;
043    }
044
045    boolean hasStreamIdentifier() {
046        return streamIdentifier;
047    }
048
049    boolean usesChecksumWithCompressedChunks() {
050        return checksumWithCompressedChunks;
051    }
052}
